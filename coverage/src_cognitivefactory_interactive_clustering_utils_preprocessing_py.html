26def preprocess( 

27 dict_of_texts: Dict[str, str], 

28 apply_stopwords_deletion: bool = False, 

29 apply_parsing_filter: bool = False, 

30 apply_lemmatization: bool = False, 

31 spacy_language_model: str = "fr_core_news_sm", 

32) -> Dict[str, str]: 

33 """ 

34 A method used to preprocess texts. 

35 It applies simple preprocessing (lowercasing, punctuations deletion, accents replacement, whitespace deletion). 

36 Some options are available to delete stopwords, apply lemmatization, and delete tokens according to their depth in the denpendency tree. 

37 

38 References: 

39 - _spaCy_: `Honnibal, M. et I. Montani (2017). spaCy 2 : Natural language understanding with Bloom embeddings, convolutional neural networks and incremental parsing.` 

40 - _spaCy_ language models: `https://spacy.io/usage/models` 

41 - _NLTK_: `Bird, Steven, Edward Loper and Ewan Klein (2009), Natural Language Processing with Python. O’Reilly Media Inc.` 

42 - _NLTK_ _'SnowballStemmer'_: `https://www.nltk.org/api/nltk.stem.html#module-nltk.stem.snowball` 

43 

44 Args: 

45 dict_of_texts (Dict[str,str]): A dictionary that contains the texts to preprocess. 

46 apply_stopwords_deletion (bool, optional): The option to delete stopwords. Defaults to `False`. 

47 apply_parsing_filter (bool, optional): The option to filter tokens based on dependency parsing results. If set, it only keeps `"ROOT"` tokens and their direct children. Defaults to `False`. 

48 apply_lemmatization (bool, optional): The option to lemmatize tokens. Defaults to `False`. 

49 spacy_language_model (str, optional): The spaCy language model to use if vectorizer is spacy. The model has to be installed. Defaults to `"fr_core_news_sm"`. 

50 

51 Raises: 

52 ValueError: Raises error if the `spacy_language_model` is not installed. 

53 

54 Returns: 

55 Dict[str,str]: A dictionary that contains the preprocessed texts. 

56 

57 Examples: 

58 ```python 

59 # Import. 

60 from cognitivefactory.interactive_clustering.utils.preprocessing import preprocess 

61 

62 # Define data. 

63 dict_of_texts={ 

64 "0": "Comment signaler une perte de carte de paiement ?", 

65 "1": "Quelle est la procédure pour chercher une carte de crédit avalée ?", 

66 "2": "Ma carte Visa a un plafond de paiment trop bas, puis-je l'augmenter ?", 

67 } 

68 

69 # Apply preprocessing. 

70 dict_of_preprocessed_texts = preprocess( 

71 dict_of_texts=dict_of_texts 

72 apply_stopwords_deletion=True, 

73 apply_parsing_filter=False, 

74 apply_lemmatization=False, 

75 spacy_language_model="fr_core_news_sm", 

76 ) 

77 

78 # Print results. 

79 print("Expected results", ";", {"0": "signaler perte carte paiement", "1": "procedure chercher carte credit avalee", "2": "carte visa plafond paiment l augmenter",}) 

80 print("Computed results", ":", dict_of_preprocessed_texts) 

81 ``` 

82 """ 

83 

84 # Initialize dictionary of preprocessed texts. 

85 dict_of_preprocessed_texts: Dict[str, str] = {} 

86 

87 # Initialize punctuation translator. 

88 punctuation_translator = str.maketrans( 

89 { 

90 punct: " " 

91 for punct in ( 

92 ".", 

93 ",", 

94 ";", 

95 ":", 

96 "!", 

97 "¡", 

98 "?", 

99 "¿", 

100 "…", 

101 "•", 

102 "(", 

103 ")", 

104 "{", 

105 "}", 

106 "[", 

107 "]", 

108 "«", 

109 "»", 

110 "^", 

111 "`", 

112 "'", 

113 '"', 

114 "\\", 

115 "/", 

116 "|", 

117 "-", 

118 "_", 

119 "#", 

120 "&", 

121 "~", 

122 "@", 

123 ) 

124 } 

125 ) 

126 

127 # Load vectorizer (spacy language model). 

128 try: 

129 spacy_nlp = spacy.load( 

130 name=spacy_language_model, 

131 disable=[ 

132 # "tagger", # Needed for lemmatization. 

133 # "parser", # Needed for filtering on dependency parsing. 

134 "ner", # Not needed 

135 ], 

136 ) 

137 except OSError as err: # `spacy_language_model` is not installed. 

138 raise ValueError("The `spacy_language_model` '" + str(spacy_language_model) + "' is not installed.") from err 

139 

140 # Initialize stemmer. 

141 ####stemmer = SnowballStemmer(language="french") 

142 

143 # For each text... 

144 for key, text in dict_of_texts.items(): 

145 

146 # Force string type. 

147 preprocessed_text: str = str(text) 

148 

149 # Apply lowercasing. 

150 preprocessed_text = text.lower() 

151 

152 # Apply punctuation deletion (before tokenization). 

153 preprocessed_text = preprocessed_text.translate(punctuation_translator) 

154 

155 # Apply tokenization and spaCy pipeline. 

156 tokens = [ 

157 token 

158 for token in spacy_nlp(preprocessed_text) 

159 if ( 

160 # Spaces are not allowed. 

161 not token.is_space 

162 ) 

163 and ( 

164 # Punctuation are not allowed. 

165 not token.is_punct 

166 and not token.is_quote 

167 ) 

168 and ( 

169 # If set, stopwords are not allowed. 

170 (not apply_stopwords_deletion) 

171 or (not token.is_stop) 

172 ) 

173 and ( 

174 # If set, stopwords are not allowed. 

175 (not apply_parsing_filter) 

176 or (len(list(token.ancestors)) <= 1) 

177 ) 

178 ] 

179 

180 # Apply retokenization with lemmatization. 

181 if apply_lemmatization: 

182 

183 preprocessed_text = " ".join([token.lemma_.strip() for token in tokens]) 

184 

185 # Apply retokenization without lemmatization. 

186 else: 

187 preprocessed_text = " ".join([token.text.strip() for token in tokens]) 

188 

189 # Apply accents deletion (after lemmatization). 

190 preprocessed_text = "".join( 

191 [char for char in unicodedata.normalize("NFKD", preprocessed_text) if not unicodedata.combining(char)] 

192 ) 

193 

194 # Store preprocessed text. 

195 dict_of_preprocessed_texts[key] = preprocessed_text 

196 

197 return dict_of_preprocessed_texts
